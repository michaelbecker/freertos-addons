129 WorkQueue::~WorkQueue()
130 {
131  //
132  // This dtor is tricky, because of the multiple objects in
133  // play, and the multithreaded nature of this specific object.
134  //
135 
136  //
137  // Note that we cannot flush the queue. If there are items
138  // in the queue maked freeAfterComplete, we would leak the
139  // memory.
140  //
141 
142  //
143  // Send a message that it's time to cleanup.
144  //
145  WorkItem *work = NULL;
146  WorkItemQueue->Enqueue(&work);
147 
148  //
149  // Wait until the thread has run enough to signal that it's done.
150  //
151  ThreadComplete->Take();
152 
153  //
154  // Then delete the queue and thread. Order doesn't matter here.
155  //
156  delete WorkItemQueue;
157  delete WorkerThread;
158  delete ThreadComplete;
159 }

190 void WorkQueue::CWorkerThread::Run()
191 {
192  while (true) {
193 
194  WorkItem *work;
195 
196  //
197  // Wait forever for work.
198  //
199  ParentWorkQueue->WorkItemQueue->Dequeue(&work);
200 
201  //
202  // If we dequeue a NULL item, its our sign to exit.
203  // We are being deconstructed.
204  //
205  if (work == NULL) {
206  //
207  // Exit the task loop.
208  //
209  break;
210  }
211 
212  //
213  // Else we have an item, run it.
214  //
215  work->Run();
216 
217  //
218  // If this was a dynamic, fire and forget item and we were
219  // requested to clean it up, do so.
220  //
221  if (work->FreeAfterRun()) {
222  delete work;
223  }
224  }
225 
226  //
227  // Signal the dtor that the thread is exiting.
228  //
229  ParentWorkQueue->ThreadComplete->Give();
230 }
